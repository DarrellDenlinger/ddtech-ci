# I tried using Volta (a Node version manager, replaces nvm)
# but that is really designed for one user. 
# It causes many problems to try and share that with service users.
# Because it installs a private version in ~darrell/.volta
# So I just did a normal system install 

curl -fsSL https://deb.nodesource.com/setup_22.x | sudo -E bash -
sudo apt install nodejs

# this install node and npm under /usr/bin
# which is visible to service users


# then your unit file can use 
ExecStart=/usr/bin/node
