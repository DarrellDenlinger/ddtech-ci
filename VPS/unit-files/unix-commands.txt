# service files live at
# /etc/systemd/system/sonoquii-server.service
# copy them there manually

# ################## Initial Registration ###################

# Reload systemd unit file definitions so it sees the new file
sudo systemctl daemon-reload

# Enable it to start automatically at boot
sudo systemctl enable sonoquii-server

# systemctl is like remote control for systemd
# start, stop, restart, reload, enable, status, is-enabled      

# Start it now
sudo systemctl start sonoquii-server

# Verify it's running
sudo systemctl status sonoquii-server


# ################## Operaion and Troubleshooting ###################

# Do NOT edit the nginx unit file, it would be overwritten by updates

# list all services
systemctl list-units --type=service

# look for a specific one
systemctl list-units --type=service | grep nginx

# check status of specific service
systemctl status nginx

# see the unit file for a service
systemctl cat nginx

# re-read all unit files from disk and reload in memory
# note this does NOT restart any services
sudo systemctl daemon-reload
# it would be typical to restart your service after this command
# sudo /var/www/sonoquii.com/restart-server.sh

# ################## nginx ###################
# ngnix is special in that we don't control its unit file
# we can control the operation Troubleshooting

# validate the configs 
sudo nginx -t

# reloads the configs without stopping the service
sudo systemctl reload nginx
